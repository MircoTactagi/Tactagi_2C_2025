118void InterrupcionPorUart(void) {
119 uint8_t tecla;
120 UartReadByte(UART_PC, &tecla);
121
122 switch(tecla) {
123 case 'O': case 'o':
124 DetenerMedicion = !DetenerMedicion;
125 break;
126 case 'H': case 'h':
127 FijarMedicion = !FijarMedicion;
128 break;
129 case 'I': case 'i':
130 DistanciaenIn = !DistanciaenIn;
131 break;
132 case 'M': case 'm': {
133 uint8_t *num_str_cm = UartItoa(distancia_max_cm, 10);
134 uint8_t *num_str_in = UartItoa(distancia_max_in, 10);
135
136 if (DistanciaenIn == false) {
137 UartSendString(UART_PC, (char*)num_str_cm);
138 UartSendString(UART_PC, " MAXIMO en cm\r\n");
139 LcdItsE0803Write(distancia_max_cm);
140 } else {
141 UartSendString(UART_PC, (char*)num_str_in);
142 UartSendString(UART_PC, " MAXIMO en In\r\n");
143 LcdItsE0803Write(distancia_max_in);
144 }
145
146 FijarMedicion = true;
147 vTaskDelay(pdMS_TO_TICKS(5000));
148 FijarMedicion = false;
149 break;
150 }
151 case 'F': case 'f':
152 periodo_lectura -= 100000;
153 TimerUpdatePeriod(TIMER_A, periodo_lectura);
154 break;
155 case 'S': case 's':
156 periodo_lectura += 100000;
157 TimerUpdatePeriod(TIMER_A, periodo_lectura);
158 break;
159 }
160}

167void Actualiza_leds(uint16_t distancia) {
168 LedsOffAll();
169
170 if (distancia >= 10 && distancia < 20) {
171 LedOn(LED_1);
172 } else if (distancia >= 20 && distancia < 30) {
173 LedOn(LED_1);
174 LedOn(LED_2);
175 } else if (distancia >= 30) {
176 LedOn(LED_1);
177 LedOn(LED_2);
178 LedOn(LED_3);
179 }
180}

187static void TareaMedirDistancia(void *pvParameters) {
188 while (true) {
189 ulTaskNotifyTake(pdTRUE, portMAX_DELAY);
190
191 if (!DetenerMedicion) {
192 uint16_t distancia = 0;
193
194 if (DistanciaenIn == false) {
195 distancia = HcSr04ReadDistanceInCentimeters();
196 if (distancia > distancia_max_cm) {
197 distancia_max_cm = distancia;
198 }
199 } else {
200 distancia = HcSr04ReadDistanceInInches();
201 if (distancia > distancia_max_in) {
202 distancia_max_in = distancia;
203 }
204 }
205
206 Actualiza_leds(distancia);
207
208 if (!FijarMedicion) {
209 LcdItsE0803Write(distancia);
210 }
211
212 EnviarDistanciaPorUART(distancia);
213 }
214 }
215}
